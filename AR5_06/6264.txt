        INCLUDE XH_FORGE_FORTRAN,LIST

        INCLUDE *VOUCHER_CF2,LIST
        INCLUDE *VOUCHER_CF3,LIST
******* Let's start off with a quick explanation on my storage.
******* I have 5 parallel arrays, one of which is 2D. They are:
*******
******* P_PROJ  - the project numbers.
******* P_DESC  - the 20 char descriptions.
******* P_ACT   - the INTEGER representation of the activity codes.
******* P_HRS   - the 2D one that's 7 across, one for each day.
******* CTRL    - holds Control Characters (CCs) and other fun stuff.
*******
******* These arrays run down 0:200.  Anyone that needs more room for
******* more projects should seriously consider getting some hobbies!!
******* The forms, running down 0:16, lay on top of the first 4 arrays
******* providing a neat little window to one's data.  The user cannot
******* see the CTRL array.  That's MY playground!  It holds characters
******* such as all of the daily form's Control Characters (CCs) and
******* a special one or two, like the '-' indicating an add of some
******* sort, or an 'I' meaning this project's hours has changed and I
******* had better write it to the database or something.  I should
******* also mention the LIST array (0:300).  It will hold the
******* proj/desc combo with a metrics indicator in character format
******* to be displayed when somebody does a wildcarded search for
******* projects or descriptions.  Used only with the Information
******* form.  ...Oh, yah,  FORM1 is the Daily form,  FORM2 is the
******* Info form, and FORM3 is the Weekly form.
*******
******* Some of the variables to look out for are:
******* FIRST   - index into the arrays, representing the first line
*******           displayed in the window.
******* PCNT    - The total number of projects.  1-relative!
******* PREDEF  - total of predefined prompts.  Anything <= to this
*******           will be *'ed as a one-time-only type prompt.  1-rel.
******* L1ST    - the LIST's first line of the window.
******* LCNT    - the total count of proj/desc's found in the LIST.
******* VCC     - is a string consisting of valid control characters.
******* RESEQ_PREDEF - a flag that signal that the order of the
*******                predefined prompts have been changed in some
*******                way or another.  After the user hits an E,Q or
*******                X this variable is checked.  If on, write out
*******                the new sequence of prompts.
******* TO      - the destination index of a move.
******* FROM    - the index of the line to be moved.
******* CURR_FORM - 1, 2, or 3.
*******
******* Okay, return your seats to their fully upright positions and
******* make sure your serving trays are up and locked.  We expect
******* a smooth ride, but it may get bumpy....
*******
        REAL TOT_HRS(0:6), DHOURS, WKTOT
        CHARACTER DATE*8, P_NUM(0:200)*7, LOGON*12, ANS_DATE*6
        CHARACTER FISCAL_DATE*10, WEEKDAY, DOW(7)*3, P_DESC(0:200)*20
        CHARACTER PNUM*6, PDESC*20, NAMES(0:10)*6, DSAV*3
        CHARACTER CFW*2, CDAY*3, HOURS*5, P_HRS(0:6,0:200)*5, METRIC
        CHARACTER CTRL(0:200), VCC*30, LIST(0:300)*37
        CHARACTER EMP_NUM*6, ORIG_ENUM*6, IN_WEEK*2, W*4
        INTEGER Q, UTS, PSEQ, PCNT, DAY
        INTEGER ECOD, CP6E, OPER, OVRW, EXST, SKIP, ARGN, WEEK, NCNT
        INTEGER COUNTER, P_CODE(0:200), PCODE, FIRST, IDAY, CC, FRST
        INTEGER PREDEF, TO, FROM, CMAX, CURR_FORM, LCNT, X, I, I2, I3
        INTEGER L1ST, L2, CNT3, CNTR
        LOGICAL SEQ_OKAY, CHANGED, WK, FIRST_TIME, DELETE, CMWKLY
        LOGICAL RESEQ_PREDEF, ADD, CREATOR, INWEEK, CLOSED
        EXTERNAL GET_NAME, GET_DATE, GET_DATE_FROM_UTS, TEXT_TO_UTS
        EXTERNAL SETUP, BREAK, SET_PROMPT, RESET_PROMPT, VSFFISCAL
        EXTERNAL VERIFY_WEEK, VERIFY_DAY, VERIFY_HOURS, VERIFY_PROJ
        EXTERNAL VERIFY_DESC, VERIFY_ACT, VERIFY_WEEK3, VERIFY_CMD2
        EXTERNAL VERIFY_PROJ2, VERIFY_DESC2, VERIFY_ACT2
        EXTERNAL VERIFY_WEEK_HOURS, VERIFY_CLOSED
        CHARACTER MODE_IN*2, LINE*30
        CHARACTER BGN_LIST_2*46, BGN_LIST_5_6*100
        CHARACTER BGN_LIST_LOGON*23, EARS*16, GET_ENUM*100
        CHARACTER INS_EMP_PROJ*100,  DEL_EMP_PROJ*100
        CHARACTER GET_PROMPTS*150, GET_YTDWKLY*150, GET_CMWKLY*150
        CHARACTER OLD_OR_NEW_YTD*100, OLD_OR_NEW_FW*100
        CHARACTER GET_NAMES*50, VAL_ENUM*50, CLO_CMDLY*100
        CHARACTER INS_CMDLY*100, DEL_CMDLY*100, BGN_LIST_CMDLY2*100
        CHARACTER BGN_LIST_CMDLY*85, GET_DAILY*200, BGN_LIST_ADJ*50
        CHARACTER BGN_LIST_CMWKLY*100, BGN_LIST_YTDWKLY*100
        COMMON/ERRS/ECOD,CP6E,OPER,OVRW,EXST,SKIP,ARGN
        COMMON/DATA/P_NUM,P_DESC,P_CODE,P_HRS,CTRL,FIRST
        COMMON/NAMES/NAMES,NCNT
        COMMON/WEEK/WEEK
        COMMON/DAY/IDAY,DOW
        COMMON/ALTER/CHANGED,CMAX
        COMMON/DATA2/LIST,LCNT
        COMMON/CHA/CREATOR
        COMMON/OLD_WEEK/CFW
        DOW(1) = 'MON'
        DOW(2) = 'TUE'
        DOW(3) = 'WED'
        DOW(4) = 'THU'
        DOW(5) = 'FRI'
        DOW(6) = 'SAT'
        DOW(7) = 'SUN'
        VCC = ' <>BCDFLMNPTW      .0123456789'
        FIRST_TIME = .TRUE.
        RESEQ_PREDEF = .FALSE.
        FROM = -1
        MODE_IN         = 'IN'
        BGN_LIST_2      = 'MODE=IN,SHARE=IN,CMWKLY(SHARE=ALL),'//
     &      'PSA,YTDWKLY'
        BGN_LIST_5_6    = 'SHARE=IN,EMP_PROJ(MODE=IN,SHARE='//
     &      'ALL),PSA(MODE=IN)'
        BGN_LIST_ADJ    = 'MODE=IN,SHARE=IN,EMP_PROJ(MODE=UPDATE,'//
     &      'SHARE=ALL)'
        BGN_LIST_LOGON  = 'MODE=IN,SHARE=IN,LOGONS'
        BGN_LIST_CMDLY  = 'MODE=IN,SHARE=IN,CMDLY(SHARE=ALL),'//
     &      'PSA(SHARE=ALL)'
        BGN_LIST_CMDLY2 = 'MODE=IN,SHARE=IN,CMDLY(MODE=UPDATE,'//
     &      'SHARE=ALL)'
        BGN_LIST_CMWKLY     = 'MODE=IN,SHARE=IN,CMWKLY(SHARE=ALL),'//
     &      'PSA(SHARE=ALL)'
        BGN_LIST_YTDWKLY    = 'MODE=IN,SHARE=IN,YTDWKLY(SHARE=ALL),'//
     &      'PSA(SHARE=ALL)'
        GET_ENUM        = 'SEL ENUM FROM LOGONS WHERE LOGON=#1'
        VAL_ENUM        = 'SEL ENUM FRO LOGONS WHERE ENUM=#1'
        INS_CMDLY       = 'INS INTO CMDLY VALUES( #1,#2,#3,#4,#5,#6 )'
        DEL_CMDLY       = 'DEL CMDLY WHE DHOURS=0.0 AND ENUM=#1 AND'//
     &      ' FW=#2 AND DAY=#3'
        CLO_CMDLY       = 'SEL COUNT(FW) FRO CMDLY WHE ENUM=#1 '//
     &      'AND FW=#2'
        INS_EMP_PROJ    = 'INSERT INTO EMP_PROJ VALUES( #1,#2,#3,#4 )'
        DEL_EMP_PROJ    = 'DEL EMP_PROJ WHERE ENUM=#1 AND '//
     &      'PROJECT=#2 AND ACT_CODE=#3'
        GET_PROMPTS     = 'SEL EMP_PROJ.PROJECT,DESCRIPTION,ACT_COD'//
     &      'E,SEQ,METRICS FRO EMP_PROJ,PSA WHE ENUM=#1 AND EMP_PRO'//
     &      'J.PROJECT=PSA.PROJECT ORD BY SEQ,EMP_PROJ.PROJECT'
        GET_DAILY       = 'SEL CMDLY.PROJECT,DESCRIPTION,ACT_CODE,'//
     &      'DAY,DHOURS FRO CMDLY,PSA WHE ENUM=#1 AND FW=#2 '//
     &      'AND CMDLY.PROJECT=PSA.PROJECT'
        GET_CMWKLY      = 'SEL CMWKLY.PROJECT,PSA.DESCRIPTION,'//
     &      'HOURS FROM CMWKLY,PSA WHERE ENUM=#1 AND FW=#2 AND '//
     &      'CMWKLY.PROJECT=PSA.PROJECT'
        GET_YTDWKLY     = 'SEL YTDWKLY.PROJECT,PSA.DESCRIPTION,'//
     &      'HOURS FROM YTDWKLY,PSA WHERE ENUM=#1 AND FW=#2 AND '//
     &      'YTDWKLY.PROJECT=PSA.PROJECT'
        OLD_OR_NEW_FW   = 'SEL COUNT(FW) FROM CMWKLY WHERE ENUM='//
     &      '#1 AND FW=#2'
        OLD_OR_NEW_YTD  = 'SEL COUNT(FW) FROM YTDWKLY WHERE ENUM='//
     &      '#1 AND FW=#2'
        GET_NAMES = 'SEL ACT_NAME FRO ACT_CODES'
        CALL SETUP
 
******* Get the Fiscal Date in the form of YYDOYFWQMM where YY=year,
******* DOY=day into the year (Feb. 1 is 032), FW=fiscal week,
******* Q=quarter and MM=month.
        CALL DAT$(DATE)
        CALL TEXT_TO_UTS(DATE,UTS,ECOD)
        ECOD = 0
        CALL GET_DATE_FROM_UTS(UTS,ANS_DATE)
        CALL VSFFISCAL(ANS_DATE, FISCAL_DATE, WEEKDAY)
        WEEK = ((ICHAR(FISCAL_DATE(6:6))-48)*10) +
     &      (ICHAR(FISCAL_DATE(7:7))-48)
        IDAY = ICHAR(WEEKDAY)-48
 
******* Use the fiscal date to open the current year's database.
        CALL ARES_INIT(ECOD,CP6E,OPER,OVRW,EXST,SKIP,ARGN)
        IF (ECOD.NE.0) GOTO 666
        CALL ARES_SET_OVERWRITE
        IF (ECOD.NE.0) GOTO 666
        EARS = 'EARS_'//FISCAL_DATE(1:2)//'.LADCSTAT'
        CALL ARES_INVOKE(EARS)
        IF (ECOD.NE.0) GOTO 666
        CALL GET_NAME(LOGON,IN_WEEK)
 
******* Use the account name to get the employee number.
        CALL ARES_BEGIN(BGN_LIST_LOGON)
        REPEAT 10, WHILE (ECOD.EQ.3)
            CALL ARES_BEGIN(BGN_LIST_LOGON)
10      CONTINUE
        CALL ARES_STARTEXT(GET_ENUM, Q, MODE_IN, LOGON)
        IF (ECOD.NE.0) GOTO 666
        CALL ARES_FETCH(Q,EMP_NUM)
        IF (ECOD.EQ.1) THEN
            WRITE(*,'(1X,A,A)')'Your logon must be added to the ',
     &          'Employee Master Table.  Please contact your manager.'
            GOTO 999
        ENDIF
        ORIG_ENUM = EMP_NUM
        CALL ARES_RELEASE(Q)
        CALL ARES_COMMIT
 
******* Put up the form showing the current week and day.
        CALL FORMPUT(FORM1)
        IF (FORM1.NE.0)CALL FORMERR(FORM1)
        WEEK1 = FISCAL_DATE(6:7)
        IF (WEEK1(1:1).EQ.'0') THEN
            WEEK1(1:1) = WEEK1(2:2)
            WEEK1(2:2) = ' '
        ENDIF
******* Process the value entered on the command line.
        INWEEK = .FALSE.
        IF (IN_WEEK.NE.'  ') THEN
            CFW = WEEK1
            CALL UPCASE(IN_WEEK,2)
            DO 11, I=1,2
                IF ((IN_WEEK(I:I).NE.'-').AND.
     &              (IN_WEEK(I:I).NE.'+').AND.
     &              (IN_WEEK(I:I).NE.' ').AND.
     &              ((IN_WEEK(I:I).LT.'0').OR.
     &               (IN_WEEK(I:I).GT.'9'))) GOTO 12
11          CONTINUE
            IF ((IN_WEEK.EQ.'- ').OR.(IN_WEEK.EQ.' -').OR.
     &          (IN_WEEK.EQ.'-1')) THEN
                IF (WEEK-1.GE.1) THEN
                    WEEK = WEEK - 1
                ELSE
                    GOTO 12
                ENDIF
            ELSE
                IF ((IN_WEEK.EQ.'+ ').OR.(IN_WEEK.EQ.' +').OR.
     &              (IN_WEEK.EQ.'+1')) THEN
                    IF (WEEK+1.LE.52) THEN
                        WEEK = WEEK + 1
                    ELSE
                        GOTO 12
                    ENDIF
                ELSE
                    IF ((IN_WEEK(1:1).EQ.'-') .OR.
     &                  (IN_WEEK(1:1).EQ.'+')) THEN
                        READ(IN_WEEK,'(BN,I)')I
                        IF ((WEEK+I.GE.1).AND.(WEEK+I.LE.52)) THEN
                            WEEK = WEEK + I
                        ELSE
                            GOTO 12
                        ENDIF
                    ELSE
                        READ(IN_WEEK,'(BN,I)')I
                        IF ((I.GE.1).AND.(I.LE.52)) THEN
                            WEEK = I
                        ELSE
                            GOTO 12
                        ENDIF
                    ENDIF
                ENDIF
            ENDIF
            WRITE(W,'(I)')WEEK
            WEEK1 = W
            CDAY = DOW(IDAY)
            INWEEK = .TRUE.
        ENDIF
12      DAY1 = DOW(IDAY)
        WEEK1_S = 1
        DAY1_S = 1
        CALL FORMMDFY(FORM1,FORGE_ABSEMPHASIS,FORM1_S)
        CALL FORMWRITE(FORM1,WEEK1,DAY1)
        CALL FORMCURRENT(FORM1)
        CURR_FORM = 0
 
******* Give the user something to look at...
        CALL FORMMDFY(FORM1,FORGE_ABSERROR,ERROR_MESSAGE)
        ERROR_MESSAGE = 'Initializing...'
        CALL FORMWRITE(FORM1,ERROR_MESSAGE)
        CALL FORMCURRENT(FORM1)
        CALL FORMMDFY(FORM1,FORGE_ABSNONE,ERROR_MESSAGE)
        ERROR_MESSAGE = ' '
        ERROR_MESSAGE_S = 0
 
******* Search for CREATOR privilege.
        CREATOR = .FALSE.
        CALL ARES_DISPLAY('DISPLAY PRIVS')
        OPEN (UNIT=3,FILE='***ARES_DISPLAY',STATUS='OLD')
        READ(3,'(A)')LINE
        READ(3,'(A)')LINE
        READ(3,'(A)')LINE
        READ(3,'(A)')LINE
        IF (LINE(29:29).EQ.'x') CREATOR = .TRUE.
        CLOSE(3)
 
******* Get the Activity Code names.
        CALL ARES_BEGIN( 'MODE=IN,ACT_CODES(SHARE=ALL)' )
        CALL ARES_STARTEXT( GET_NAMES, Q, 'IN' )
        CALL ARES_FETCH( Q, NAMES(0) )
        NCNT = 1
        REPEAT 20, WHILE (ECOD.EQ.0)
            CALL ARES_FETCH( Q, NAMES(NCNT) )
            NCNT = NCNT + 1
20      CONTINUE
        NCNT = NCNT - 1
        CALL ARES_RELEASE( Q )
        CALL ARES_COMMIT
 
******* Possibly change the Employee Number.
        GOTO 16
15      IF (CREATOR) THEN
            WKTOT1 = EMP_NUM
            ERROR_MESSAGE = 'Enter a new Emp-Num in the Week Total '//
     &          'field below.'
            CALL FORMWRITE(FORM1,ERROR_MESSAGE,WKTOT1)
            CALL FORMMDFY(FORM1,FORGE_ABSEMPHASIS,WKTOT1)
            CALL FORMREAD(FORM1,WKTOT1)
            IF (WKTOT1.EQ.'      ') THEN
                EMP_NUM = ORIG_ENUM
            ELSE
                IF (WKTOT1.NE.EMP_NUM) THEN
                    CALL ARES_BEGIN( BGN_LIST_LOGON )
                    IF (ECOD.NE.0) GOTO 666
                    CALL ARES_STARTEXT(VAL_ENUM,Q,MODE_IN,WKTOT1)
                    IF (ECOD.NE.0) GOTO 666
                    CALL ARES_FETCH( Q, NAMES(10) )
                    NAMES(10) = ' '
                    IF (ECOD.EQ.1) THEN
                        WKTOT1 = ' '
                        ERROR_MESSAGE = ' '
                        CALL FORMWRITE(FORM1,ERROR_MESSAGE,WKTOT1)
                        CALL ARES_RELEASE( Q )
                        CALL ARES_COMMIT
                        GOTO 15
                    ENDIF
                    EMP_NUM = WKTOT1
                    WKTOT1 = ' '
                    CALL ARES_RELEASE( Q )
                    CALL ARES_COMMIT
                ENDIF
            ENDIF
        ENDIF
16      CONTINUE
******* Clear the arrays in preparation for retrieving the prompts
        DO 40, I=0,200
            P_NUM(I) = ' '
            P_DESC(I) = ' '
            P_CODE(I) = 0
            CTRL(I) = ' '
            DO 30, J=0,6
                P_HRS(J,I) = '  .  '
30          CONTINUE
40      CONTINUE
        CALL ARES_BEGIN( BGN_LIST_5_6 )
        IF (ECOD.NE.0) GOTO 666
        CALL ARES_STARTEXT(GET_PROMPTS, Q, MODE_IN, EMP_NUM)
        IF (ECOD.NE.0) GOTO 666
        CALL ARES_FETCH( Q, PNUM, PDESC, PCODE, PSEQ, METRIC)
        PCNT = 0
        SEQ_OKAY = .TRUE.
        REPEAT 60, WHILE (ECOD.EQ.0)
            IF (P_NUM(PSEQ).NE.' ') SEQ_OKAY = .FALSE.
            IF (.NOT.SEQ_OKAY) THEN
                PSEQ = PCNT
                CTRL(PSEQ) = 'A'
            ENDIF
            P_NUM(PSEQ) = PNUM
            P_DESC(PSEQ) = PDESC
            P_CODE(PSEQ) = PCODE
            PCNT = PCNT + 1
*******     Check if Metric and not broken out.
            IF ((METRIC.EQ.'Y').AND.(PCODE.EQ.0)) THEN
                CTRL(PSEQ) = 'D'
                DO 50, I=1,NCNT-1
                    P_NUM(PSEQ+I) = PNUM
                    P_DESC(PSEQ+I) = PDESC
                    P_CODE(PSEQ+I) = I
                    CTRL(PSEQ+I) = 'A'
50              CONTINUE
                PCNT = PCNT + NCNT-1
                SEQ_OKAY = .FALSE.
            ENDIF
            CALL ARES_FETCH( Q, PNUM, PDESC, PCODE, PSEQ, METRIC)
60      CONTINUE
        CALL ARES_RELEASE(Q)
        CALL ARES_COMMIT
 
******* Adjust the list, deleting out the Metric entries with no
******* activity codes, and overwriting the prompts w/bad sequence #s
        CALL ARES_BEGIN( BGN_LIST_ADJ )
        IF (ECOD.NE.0) GOTO 666
        DO 70, I=0,PCNT-1
            IF (CTRL(I).EQ.'D') THEN
                CALL ARES_EXECTEXT(DEL_EMP_PROJ,EMP_NUM,P_NUM(I)(:6),
     &              P_CODE(I))
                IF (ECOD.NE.0) GOTO 666
                CALL SCOOCH( I, PCNT, -1 )
            ENDIF
70      CONTINUE
        DO 80, I=0,PCNT-1
            IF (CTRL(I).EQ.'A') THEN
                CALL ARES_EXECTEXT(INS_EMP_PROJ,EMP_NUM,P_NUM(I)(:6),
     &              P_CODE(I),I)
                IF (ECOD.NE.0) GOTO 666
            ENDIF
80      CONTINUE
        PREDEF = PCNT
******* At this point, PREDEF is the count of predefined prompts.
******* PCNT will change to reflect the total count of projects.
******* Both of these counts are 1 relative.
        CALL ARES_COMMIT
 
        FIRST = 0
******* If a week was specified in the command line, see if it is closed.
        IF (INWEEK) GOTO 260
******* Clear all the hours on the form.
90      DO 110, I=0,6
            TOT_HRS(I) = 0.0
            DO 100, J=0,200
                P_HRS(I,J) = '  .  '
100         CONTINUE
110     CONTINUE
 
        CALL FORMMDFY(FORM1,FORGE_ABSERROR,ERROR_MESSAGE)
        IF (CLOSED) THEN
            ERROR_MESSAGE='Getting data for closed fiscal week '//
     &          WEEK1//'...'
        ELSE
            ERROR_MESSAGE='Getting data for fiscal week '//WEEK1//'...'
        ENDIF
        CALL FORMWRITE(FORM1,ERROR_MESSAGE)
        CALL FORMCURRENT(FORM1)
        CALL FORMMDFY(FORM1,FORGE_ABSNONE,ERROR_MESSAGE)
        ERROR_MESSAGE = ' '
        ERROR_MESSAGE_S = 0
 
******* Get the hours already vouchered for the current week.
        CALL ARES_BEGIN( BGN_LIST_CMDLY )
        IF (ECOD.NE.0) GOTO 666
        CALL ARES_STARTEXT(GET_DAILY,Q,MODE_IN,EMP_NUM,WEEK1)
        IF (ECOD.NE.0) GOTO 666
        CALL ARES_FETCH( Q, PNUM, PDESC, PCODE, DAY, HOURS )
        REPEAT 140, WHILE (ECOD.EQ.0)
            I2=0
*******     Look through the prompts for a match.  If you run into
*******     a blank entry, star (*) the entry as non-predefined.
            REPEAT 120, WHILE ((P_NUM(I2)(1:6).NE.PNUM(1:6)).OR.
     &          (P_CODE(I2).NE.PCODE))
                IF (P_NUM(I2).EQ.'     ') THEN
                    P_NUM(I2) = PNUM
                    P_NUM(I2)(7:7) = '*'
                    P_DESC(I2) = PDESC
                    P_CODE(I2) = PCODE
                    PCNT = PCNT + 1
                    GOTO 130
                ENDIF
                I2 = I2 + 1
120         CONTINUE
*******     ARES returns '8    ', so add a decimal point and zeros.
130         IF (INDEX(HOURS,'.').EQ.0)
     &          HOURS(INDEX(HOURS,' '):) = '.00'
*******     Read the character data into a real, then write it back
*******     into the character in order to get the ' x.xx' format.
            READ(HOURS,'(BN,F5.2)')DHOURS
            WRITE(HOURS,'(F5.2)')DHOURS
            IF (HOURS(5:5).EQ.' ') HOURS(5:5) = '0'
*******     Load the hours array, update the total array, and mark
*******     that particular total to be written to the screen.
            P_HRS(DAY-1,I2) = HOURS
            TOT_HRS(DAY-1) = TOT_HRS(DAY-1) + DHOURS
            TOTAL1_S(DAY-1) = 1
            CALL ARES_FETCH( Q, PNUM, PDESC, PCODE, DAY, HOURS )
140     CONTINUE
******* Write the totals to the screen.
        CALL ARES_RELEASE( Q )
        WKTOT = 0.0
        DO 150, I=0,6
            WRITE(TOTAL1(I),'(F5.2)')TOT_HRS(I)
            WKTOT = WKTOT + TOT_HRS(I)
150     CONTINUE
        WRITE(WKTOT1,'(F6.2)')WKTOT
        WKTOT1_S = 1
        CALL ARES_COMMIT
 
******* First time through this form, put up some info.
        IF ((FIRST_TIME).AND.(.NOT.CLOSED)) THEN
            FIRST_TIME = .FALSE.
            ERROR_MESSAGE = 'E, Q, or X in the DAY field will '//
     &          'exit.  A ? in any field gives information.'
        ENDIF
        CALL FORMWRITE(FORM1,ERROR_MESSAGE)
        ERROR_MESSAGE_S = 0
 
******* If coming from another form, define the verification routines.
        IF (CURR_FORM.NE.1) THEN
            CURR_FORM = 1
            CALL FORMVRFY(FORM1,VERIFY_WEEK,WEEK1)
            CALL FORMVRFY(FORM1,VERIFY_DAY,DAY1)
            DO 170, J=0,16
                CALL FORMVRFY(FORM1,VERIFY_PROJ,PROJ1(J))
                CALL FORMVRFY(FORM1,VERIFY_DESC,DESC1(J))
                CALL FORMVRFY(FORM1,VERIFY_ACT,ACT1(J))
                DO 160, I=0,6
                    CALL FORMVRFY(FORM1,VERIFY_HOURS,HOURS1(I,J))
160             CONTINUE
170         CONTINUE
        ENDIF
******* Make sure at least one line is at the top of the screen.
180     IF (FIRST.GE.PCNT) FIRST = PCNT - 1
        IF (FIRST.LT.0) FIRST = 0
        CALL LOAD_DATA(1, FIRST)
        CALL FORMWRITE(FORM1,FORM1_S)
        DO 190, I=0,16
            IF (HOURS1_S(IDAY-1,I).NE.0) HOURS1_S(IDAY-1,I)=0
190     CONTINUE
        CALL FORMMDFY(FORM1,FORGE_ABSNONE,FORM1_S)
        CALL FORMWRITE(FORM1,ERROR_MESSAGE)
        CALL FORMCURRENT(FORM1)
******* Remember the current week and day.
        CFW = WEEK1
200     CDAY = DAY1
******* Find the integer representation of the day (1=MON).
        DO 210, IDAY=1,7
            IF (DOW(IDAY).EQ.CDAY) GOTO 220
210     CONTINUE
220     CONTINUE
        IF (CLOSED) THEN
            ERROR_MESSAGE = 'This is a CLOSED week.  Only scrolling'//
     &          ' CCs may be used.'
            CALL FORMWRITE(FORM1,ERROR_MESSAGE)
            ERROR_MESSAGE = ' '
            ERROR_MESSAGE_S = 0
            DO 225, I=0,MIN(16,PCNT-FIRST-1)
                CALL FORMVRFY(FORM1,VERIFY_CLOSED,HOURS1(IDAY-1,I))
225         CONTINUE
        ENDIF
        CALL FORMZAPS( FORM1 )
******* Set the selection array to turn on Week, Day, day's Hours,
******* and any blank lines on the form.

                      READ(HOURS3(CNT3),'(BN,F5.2)')DHOURS

1230            CONTINUE

                PROJ1_S(I-FIRST) = 1

1231            CONTINUE

*******                         of the Cmd column is over.
                                IF (CMD2(I3).EQ.'G') THEN
                                  PROJ2(I2-FIRST)(1:6) = LIST(I3+L1ST)
                                  DESC2(I2-FIRST) = LIST(I3+L1ST)(10:29)
                                  CALL FORMWRITE(FORM2,PROJ2(I2-FIRST),
     &                              DESC2(I2-FIRST),ACT2(I2-FIRST))
                                    LIST(0) = LIST(I3+L1ST)
                                    LCNT = 1
                                  CALL LOAD_LIST(-1)
                                  GOTO 1420
                                ELSE
                                  IF ((CMD2(I3).EQ.'B') .OR.
     &                              (CMD2(I3).EQ.'N') .OR.
     &                              (CMD2(I3).EQ.'P') .OR.
     &                              (CMD2(I3).EQ.'T')) THEN
                                    IF (CMD2(I3).EQ.'B')
     &                                L2 = MAX(LCNT-16,0)
                                    IF (CMD2(I3).EQ.'T') L2=0
                                    IF (CMD2(I3).EQ.'N') THEN
                                      L2 = MIN(LCNT-1,L1ST+17)
                                      IF (L2.LT.0) L2=0
                                    ENDIF
                                    IF (CMD2(I3).EQ.'P')
     &                                L2 = MAX(0,L1ST-17)
                                    IF (L2.NE.L1ST) THEN
                                      CALL LOAD_LIST(-1)
                                      L1ST = L2
                                    ENDIF
                                    GOTO 1479
                                  ENDIF
                                ENDIF
                              ENDIF
1490                        CONTINUE
*******                     Clear the Cmd and Info fields from the form.
*******                     If we've reached this point, we assume the
*******                     user didn't find what they were looking
*******                     for, so clear and return to the top of the
*******                     CURR_FORM.eq.2 loop.
                            CALL LOAD_LIST(-1)
                            PROJ2(I2-FIRST) = ' '
                            DESC2(I2-FIRST) = ' '
                            ACT2(I2-FIRST) = ' '
                            CALL FORMWRITE(FORM2,PROJ2(I2-FIRST),
     &                        DESC2(I2-FIRST),ACT2(I2-FIRST))
                            CALL FORMCURRENT(FORM2)
                            GOTO 1419
                          ENDIF
                        ENDIF
                    ENDIF
*******             Turn off the rev-vid.
                    PROJ2_S(I2-FIRST) = 1
                    DESC2_S(I2-FIRST) = 1
                    ACT2_S(I2-FIRST) = 1
                    CALL FORMMDFY(FORM2,FORGE_ABSNONE,FORM2_S)
                    CALL FORMWRITE(FORM2,FORM2_S)
                    CALL FORMCURRENT(FORM2)
                    PROJ2_S(I2-FIRST) = 0
                    DESC2_S(I2-FIRST) = 0
                    ACT2_S(I2-FIRST) = 0
*******             Check if they've cleared the proj/desc/act.
                    IF ((PROJ2(I2-FIRST)(:6).EQ.'      ') .AND.
     &                  (DESC2(I2-FIRST).EQ.'                    ').AND.
     &                  (ACT2(I2-FIRST).EQ.'     ')) THEN
*******                 Delete the line.
                        CALL SCOOCH(I2,PCNT,-1)
                        IF (I2.LT.PREDEF) THEN
                            RESEQ_PREDEF = .TRUE.
                            PREDEF = PREDEF - 1
                        ENDIF
                        CALL LOAD_DATA(2,FIRST)
                        CALL FORMWRITE(FORM2,FORM2_S)
                        CALL FORMCURRENT(FORM2)
                        IF (CTRL(I2).EQ.'-') GOTO 1321
                    ELSE
*******                 Replaces the pointer with a * if needed.
                        IF (PROJ2(I2-FIRST)(7:7).EQ.'>') THEN
                            IF (I2.GE.PREDEF) THEN
                                PROJ2(I2-FIRST)(7:7) = '*'
                                P_NUM(I2)(7:7) = '*'
                            ELSE
                                PROJ2(I2-FIRST)(7:7) = ' '
                            ENDIF
                            CALL FORMWRITE(FORM2,PROJ2(I2-FIRST))
                            CALL FORMCURRENT(FORM2)
                        ENDIF
                    ENDIF
                ENDIF
            ENDIF
*******     There!  That wasn't so bad, was it??
1330    CONTINUE
******* If an ALL was specified one of the acts from above, some of
******* lines to be added may have been scrolled off the screen.  This
******* next loop will clear any pending adds, since the user would
******* have to reset his window anyway.
        DO 1352 I2=PCNT-1,FIRST+16,-1
            IF (CTRL(I2).EQ.'-') THEN
                CALL SCOOCH(I2,PCNT,-1)
                IF (I2.LT.PREDEF) THEN
                    PREDEF = PREDEF - 1
                    RESEQ_PREDEF = .TRUE.
                ENDIF
            ENDIF
1352    CONTINUE
******* Go back to the Daily form if you're not there already.
        IF (CURR_FORM.NE.1) THEN
            CALL FORMRLS(FORM2)
            CALL FORMPUT(FORM1)
            WEEK1 = WEEK2
            DAY1 = DSAV
            CALL FORMMDFY(FORM1,FORGE_ABSEMPHASIS,WEEK1,DAY1)
            CALL FORMWRITE(FORM1,WEEK1,DAY1)
            CALL FORMCURRENT(FORM1)
            WKTOT = 0.0
*******     The LOAD_DATA routine doesn't recalc the totals, so do it.
            DO 1353, I2=0,6
                TOT_HRS(I2) = 0.0
                DO 1355, I3=0,PCNT-1
                    READ(P_HRS(I2,I3),'(BN,F5.2)')DHOURS
                    TOT_HRS(I2) = TOT_HRS(I2) + DHOURS
1355            CONTINUE
                WRITE(TOTAL1(I2),'(F5.2)')TOT_HRS(I2)
                WKTOT = WKTOT + TOT_HRS(I2)
1353        CONTINUE
            WRITE(WKTOT1,'(F6.2)')WKTOT
            CALL LOAD_DATA(1,FIRST)
            CALL FORMWRITE(FORM1,FORM1_S)
            CALL FORMCURRENT(FORM1)
            CURR_FORM = 1
            CALL FORMVRFY(FORM1,VERIFY_WEEK,WEEK1)
            CALL FORMVRFY(FORM1,VERIFY_DAY,DAY1)
            DO 1350, J=0,16
                CALL FORMVRFY(FORM1,VERIFY_PROJ,PROJ1(J))
                CALL FORMVRFY(FORM1,VERIFY_DESC,DESC1(J))
                CALL FORMVRFY(FORM1,VERIFY_ACT,ACT1(J))
                DO 1340, I2=0,6
                    CALL FORMVRFY(FORM1,VERIFY_HOURS,HOURS1(I2,J))
1340            CONTINUE
1350        CONTINUE
        ENDIF
        GOTO 325
 
666     CONTINUE
******* This is where all bad ARES_ calls are directed to.
        CALL ARES_ERRMSG
999     CONTINUE
******* This is the code that writes your altered predefs back to the
******* database.  I do it at the very end so that if the program
******* screws up the predefs at some point, one can just <Ctrl-Y> to
******* avoid writing trashed prompt list to the db.  (OF COURSE, this
******* will NEVER happen now....)
        IF (RESEQ_PREDEF) THEN
            CALL ARES_BEGIN( BGN_LIST_ADJ )
            IF (ECOD.NE.0) GOTO 666
            DO 998, I=0,PREDEF-1
                CALL ARES_EXECTEXT(INS_EMP_PROJ,EMP_NUM,P_NUM(I)(:6),
     &              P_CODE(I),I)
                IF (ECOD.NE.0) GOTO 666
998         CONTINUE
            CALL ARES_COMMIT
        ENDIF
        RETURN
        END
 
        SUBROUTINE VERIFY_WEEK( WEEK, FNUM, ERR_TEXT )
        INCLUDE *VOUCHER_CF1
        CHARACTER WEEK*4, ERR_TEXT*79
        INTEGER FNUM, WNUM, CURWK
        COMMON/WEEK/CURWK
        ERR_TEXT = ' '
        IF (WEEK(1:1).EQ.'?') THEN
            WRITE(WEEK,'(I)')CURWK
            WRITE(WEEK1,'(I)')CURWK
            CALL FORMWRITE(FORM1,WEEK1)
            ERR_TEXT = 'Enter an integer in the range of 1 through '//
     &          '52 for the fiscal week, -, or +.'
            RETURN
        ENDIF
        IF (INDEX('EeQqXx',DAY1(1:1)).NE.0) RETURN
        IF (WEEK(1:1).EQ.'-')
     &      WRITE(WEEK,'(I)')CURWK-1
        IF (WEEK(1:1).EQ.'+')
     &      WRITE(WEEK,'(I)')CURWK+1
        IF ((((WEEK(1:1).LT.'0').AND.(WEEK(1:1).NE.' ')).OR.
     &      (WEEK(1:1).GT.'9')) .OR.
     &      (((WEEK(2:2).LT.'0').AND.(WEEK(2:2).NE.' ')).OR.
     &      (WEEK(2:2).GT.'9'))) THEN
            ERR_TEXT = 'Invalid characters in WEEK field.'
            RETURN

        READ(WEEK,'(BN,I)')WNUM
        IF ((WNUM.GT.52).OR.(WNUM.LT.1)) THEN
            ERR_TEXT = 'WEEK value out of range.'
        ENDIF
        WRITE(WEEK,'(I)')WNUM
        RETURN
        END

        SUBROUTINE VERIFY_DAY( DAY, FNUM, ERR_TEXT )
        INCLUDE *VOUCHER_CF1
        CHARACTER DAY*4, ERR_TEXT*79, DOW(7)*3, DAYS*5
        INTEGER FNUM, IDAY, X
        LOGICAL CREATOR
        COMMON/CHA/CREATOR
        COMMON/DAY/IDAY,DOW
        ERR_TEXT = ' '
        CALL UPCASE( DAY, 3 )
        IF (DAY(1:1).EQ.'?') THEN
            DAY(1:1) = ' '
            DAY1 = DOW(IDAY)
            CALL FORMWRITE(FORM1,DAY1)
            ERR_TEXT = 'Enter Mon, TUe, Wed'//
     &          ', THu, Fri, SAt, SUn, -, +, E, Q, or X.'
            RETURN
        ENDIF
        IF ((CREATOR).AND.(DAY.EQ.'CHA')) RETURN
        IF ((DAY(1:1).EQ.'E').OR.(DAY(1:1).EQ.'Q').OR.
     &      (DAY(1:1).EQ.'X')) THEN
            DAY = 'END'
            RETURN
        ENDIF
        IF (DAY(1:1).EQ.'-') THEN
            IF (IDAY.EQ.1) THEN
                DAY = 'SUN'
            ELSE
                DAY = DOW(IDAY-1)
            ENDIF
            RETURN
        ENDIF
        IF (DAY(1:1).EQ.'+') THEN
            IF (IDAY.EQ.7) THEN
                DAY = 'MON'
            ELSE
                DAY = DOW(IDAY+1)
            ENDIF
            RETURN
        ENDIF
        DO 10, I=1,3
            IF ((DAY(I:I).LT.'A').OR.(DAY(I:I).GT.'Z')) THEN
                ERR_TEXT = 'Invalid characters in DAY field.'
                RETURN
            ENDIF
10      CONTINUE
        DAYS = 'MTWFS'
        X = INDEX(DAYS,DAY(1:1))
        GOTO (30,40,50,60,70),X
        ERR_TEXT = ''''//DAY(1:3)//''' is not a valid day.'
        RETURN
30      CONTINUE
        DAY1(1:3) = 'MON'
        GOTO 100
40      CONTINUE
        IF (DAY(2:2).EQ.'H') THEN
            DAY1(1:3) = 'THU'
        ELSE
            DAY1(1:3) = 'TUE'
        ENDIF
        GOTO 100
50      CONTINUE
        DAY1(1:3) = 'WED'
        GOTO 100
60      CONTINUE
        DAY1(1:3) = 'FRI'
        GOTO 100
70      CONTINUE
        IF (DAY(2:2).EQ.'U') THEN
            DAY1(1:3) = 'SUN'
        ELSE
            DAY1(1:3) = 'SAT'
        ENDIF
100     CONTINUE
        CALL FORMWRITE(FORM1,DAY1)
        RETURN
        END
 
        SUBROUTINE VERIFY_HOURS( HOURS, HNUM, ERR_TEXT )
        INCLUDE *VOUCHER_CF1
        CHARACTER HOURS*5, ERR_TEXT*79, VCC*30, HR*4
        INTEGER HNUM, INDX, STRT, ARYINDX, CMAX, DEC
        INTEGER IDAY, DOW
        REAL DHOURS
        LOGICAL CHANGED, CLR
        COMMON/ALTER/CHANGED,CMAX
        COMMON/DAY/IDAY,DOW
        VCC = '?<>BCDFLMNPTW      .0123456789'
        ERR_TEXT = ' '
        CALL UPCASE( HOURS, 1 )
        ARYINDX = (HNUM-HOURS1_N) / 7
        IF (ARYINDX.GT.CMAX) CMAX=ARYINDX
        IF (HOURS(1:1).EQ.'?') THEN
            HOURS(1:1) = ' '
            HOURS1(IDAY-1,ARYINDX)(1:1) = ' '
            CALL FORMWRITE(FORM1,HOURS1(IDAY-1,ARYINDX))
            ERR_TEXT = '<before  >after  Bottom  Center  Del '//
     &          ' First  Last  Move  Next  Prev  Top  Week'
            RETURN
        ENDIF
        IF ((HOURS(1:1).GE.'a').AND.(HOURS(1:1).LE.'z'))
     &      HOURS(1:1) = CHAR(ICHAR(HOURS(1:1))-32)
        INDX = INDEX(VCC,HOURS(1:1))
        IF (INDX.EQ.0) THEN
            ERR_TEXT = 'Invalid characters entered.  Enter a ? '//
     &          'for a list of valid control characters.'
            RETURN
        ENDIF
        IF ((PROJ1(ARYINDX)(1:6).EQ.'      ').AND.
     &      (DESC1(ARYINDX)(1:6).EQ.'      ')) THEN
            HOURS = '  .  '
            HOURS1(IDAY-1,ARYINDX) = '  .  '
            CALL FORMWRITE(FORM1,HOURS1(IDAY-1,ARYINDX))
            RETURN
        ENDIF
        STRT = 1
        IF (INDX.LT.15) STRT = 2
        DO 10, I=STRT,5
            IF (INDEX(VCC(19:),HOURS(I:I)).EQ.0) THEN
                ERR_TEXT = 'Invalid characters in hours field.'
                RETURN
            ENDIF
10      CONTINUE
        DEC = INDEX(HOURS,'.')
        IF ((DEC.GT.0).AND.(INDEX(HOURS(DEC+1:),'.').NE.0)) THEN
            ERR_TEXT = 'Too many decimal points.  Please re-enter.'
            RETURN
        ENDIF
        IF (STRT.EQ.2) THEN
            HR = HOURS(2:5)
            READ(HR,'(BN,F4.2)')DHOURS
        ELSE
            READ(HOURS,'(BN,F5.2)')DHOURS
        ENDIF
        IF (INDX.EQ.6) THEN
            CLR = .TRUE.
            DO 20, I=0,6
                IF (HOURS1(I,ARYINDX)(2:5).NE.' .  ') THEN
                    CLR = .FALSE.
                    GOTO 30
                ENDIF
20          CONTINUE
30          IF (.NOT.CLR) THEN
                HOURS(1:1) = ' '
                HOURS1(IDAY-1,ARYINDX)(1:1) = ' '
                CALL FORMWRITE(FORM1,HOURS1(IDAY-1,ARYINDX))
                ERR_TEXT = 'Use the Week CC to clear the hours for '//
     &              'this project before Deleting it.'
                RETURN
            ENDIF
        ENDIF
        CHANGED = .TRUE.
        IF (DHOURS.EQ.0.0) THEN
            HOURS(2:) = ' .  '
            IF (STRT.EQ.1) HOURS(1:1) = ' '
            RETURN
        ENDIF
        IF (INDEX(HOURS(STRT:5),'.').EQ.0) DHOURS=DHOURS*100
        IF (DHOURS.GE.10) STRT = 1
        IF (DHOURS.GT.99) THEN
!           ERR_TEXT = 'Gimme a break!  There are only 24 hours in '//
!    &          'a day!'
            ERR_TEXT = 'This number is too big to be displayed.  '//
     &          'Please enter a number less than 99.'
            RETURN
        ENDIF
        IF (STRT.EQ.2) THEN
            WRITE(HR,'(F4.2)')DHOURS
            HOURS(2:5) = HR
        ELSE
             WRITE(HOURS,'(F5.2)')DHOURS
        ENDIF
        IF ((HOURS(4:5).NE.'00').AND.(HOURS(4:5).NE.'25').AND.
     &      (HOURS(4:5).NE.'50').AND.(HOURS(4:5).NE.'75')) THEN
            ERR_TEXT = 'Only quarter hour increments are allowed. '//
     &          ' ( .00, .25, .50, .75 )'
            RETURN
        ENDIF
        RETURN
        END

            ACT2(ARYINDX)(1:2) = '  '
            CALL FORMWRITE(FORM2,ACT2(ARYINDX))
            ERR_TEXT = 'Enter ''ALL'' to get all possible Metrics '//
     &          'Activity codes.'
            RETURN
        ENDIF
        IF (ACT(1:1).EQ.'A') THEN
            ACT2(ARYINDX) = 'ALL  '
            CALL FORMWRITE(FORM2,ACT2(ARYINDX))
            RETURN
        ENDIF
        IF (ACT(1:1).EQ.'C') THEN
            P_CODE(ARYINDX+FIRST) = 3
            ACT2(ARYINDX) = 'CODE '
            CALL FORMWRITE(FORM2,ACT2(ARYINDX))
            RETURN
        ENDIF
        IF ((PROJ2(ARYINDX)(1:6).EQ.'      ') .AND.
     &      (DESC2(ARYINDX)(1:6).EQ.'      ')) THEN
            ACT = ' '
            ACT2(ARYINDX) = ' '
            CALL FORMWRITE(FORM2,ACT2(ARYINDX))
            RETURN
        ENDIF
        DO 10, I=0,NCNT-1
            IF (ACT(1:2).EQ.NAMES(I)(1:2)) THEN
                P_CODE(ARYINDX+FIRST) = I
                ACT2(ARYINDX)(1:5) = NAMES(I)(1:5)
                CALL FORMWRITE(FORM2,ACT2(ARYINDX))
                RETURN
            ENDIF
10      CONTINUE
        ACT2(ARYINDX) = ' '
        CALL FORMWRITE(FORM2,ACT2(ARYINDX))
        ERR_TEXT = 'Invalid Activity.  Choose from:  All SPecs '//
     &      'DEsgn Code TEst DOcum SUprt TRain'
        RETURN
        END
 
        SUBROUTINE SCOOCH( INDEX, TOTAL, OFFSET )
******* This routine shifts the list around, either deleting one
******* line (OFFSET=-1) or creating X blank lines (OFFSET=X).  It
******* will adjust the TOTAL passed in accordingly.
        INTEGER INDEX, TOTAL, OFFSET, P_CODE(0:200), FIRST
        CHARACTER P_NUM(0:200)*7, P_DESC(0:200)*20
        CHARACTER P_HRS(0:6,0:200)*5, CTRL(0:200)
        COMMON/DATA/P_NUM,P_DESC,P_CODE,P_HRS,CTRL,FIRST
        IF (OFFSET.GT.0) THEN
*******     This next check was never implemented.
*******     IF (TOTAL+OFFSET.GT.200) THEN
*******         OFFSET = 0
*******         RETURN
*******     ENDIF
            DO 10, I=TOTAL,INDEX,-1
                P_NUM(I+OFFSET) = P_NUM(I)
                P_NUM(I) = ' '
                P_DESC(I+OFFSET) = P_DESC(I)
                P_DESC(I) = ' '
                P_CODE(I+OFFSET) = P_CODE(I)
                P_CODE(I) = 0
                CTRL(I+OFFSET) = CTRL(I)
                IF (CTRL(I+OFFSET).EQ.' ') CTRL(I+OFFSET) = 'A'
                CTRL(I) = ' '
                DO 5 J=0,6
                    P_HRS(J,I+OFFSET) = P_HRS(J,I)
                    P_HRS(J,I) = '  .  '
5               CONTINUE
10          CONTINUE
            TOTAL = TOTAL + OFFSET
        ELSE
            DO 20, I=INDEX,TOTAL-1
                P_NUM(I) = P_NUM(I+1)
                P_NUM(I+1) = ' '
                P_DESC(I) = P_DESC(I+1)
                P_DESC(I+1) = ' '
                P_CODE(I) = P_CODE(I+1)
                P_CODE(I+1) = 0
                CTRL(I) = CTRL(I+1)
                IF (CTRL(I).EQ.' ') CTRL(I) = 'A'
                CTRL(I+1) = ' '
                DO 15 J=0,6
                    P_HRS(J,I) = P_HRS(J,I+1)
                    P_HRS(J,I+1) = '  .  '
15              CONTINUE
20          CONTINUE
            TOTAL = TOTAL - 1
        ENDIF
        RETURN
        END
 
        SUBROUTINE LOAD_DATA (F, FIRST)
******* This routine loads the appropriate form with data from the
******* P_ arrays with respect to the FIRST variable.
        INTEGER F, FIRST, P_CODE(0:200), NCNT, FRST
        CHARACTER P_NUM(0:200)*7, P_DESC(0:200)*20
        CHARACTER NAMES(0:10)*6, P_HRS(0:6,0:200)*5, CTRL(0:200)
        COMMON/DATA/P_NUM,P_DESC,P_CODE,P_HRS,CTRL,FRST
        COMMON/NAMES/NAMES,NCNT
        INCLUDE *VOUCHER_CF1
        INCLUDE *VOUCHER_CF2
        IF (F.EQ.1) THEN
            CALL FORMZAPS( FORM1 )
            DO 10, I=FIRST, FIRST+16
                IF (PROJ1(I-FIRST).NE.P_NUM(I)) THEN
                    PROJ1(I-FIRST) = P_NUM(I)
                    PROJ1_S(I-FIRST) = 1
                ENDIF
                IF (DESC1(I-FIRST).NE.P_DESC(I)) THEN
                    DESC1(I-FIRST) = P_DESC(I)
                    DESC1_S(I-FIRST) = 1
                ENDIF
                IF ((ACT1(I-FIRST).NE.NAMES(P_CODE(I))).OR.
     &              (DESC1_S(I-FIRST).EQ.1)) THEN
                    ACT1(I-FIRST) = NAMES(P_CODE(I))
                    ACT1_S(I-FIRST) = 1
                ENDIF
                DO 9, J=0,6
                    IF (HOURS1(J,I-FIRST) .NE. P_HRS(J,I)) THEN
                        HOURS1(J,I-FIRST) = P_HRS(J,I)
                        HOURS1_S(J,I-FIRST) = 1
                    ENDIF
9               CONTINUE
                IF (I-FIRST.LE.6) TOTAL1_S(I-FIRST) = 1
10          CONTINUE
            WKTOT1_S = 1
        ENDIF
        IF (F.EQ.2) THEN
            CALL FORMZAPS( FORM2 )
            DO 20, I=FIRST, FIRST+16
                IF (PROJ2(I-FIRST).NE.P_NUM(I)) THEN
                    PROJ2(I-FIRST) = P_NUM(I)
                    PROJ2_S(I-FIRST) = 1
                ENDIF
                IF (DESC2(I-FIRST).NE.P_DESC(I)) THEN
                    DESC2(I-FIRST) = P_DESC(I)
                    DESC2_S(I-FIRST) = 1
                ENDIF
                IF ((ACT2(I-FIRST).NE.NAMES(P_CODE(I))).OR.
     &              (DESC2_S(I-FIRST).EQ.1)) THEN
                    ACT2(I-FIRST) = NAMES(P_CODE(I))
                    ACT2_S(I-FIRST) = 1
                ENDIF
20          CONTINUE
        ENDIF
100     RETURN
        END
 
******* This converts input lines to uppercase.
        SUBROUTINE UPCASE( STRING, LEN )
        CHARACTER STRING*40
        INTEGER LEN, LCV
        DO 10, LCV=1,LEN
            IF ((STRING(LCV:LCV).GE.'a') .AND.
     &          (STRING(LCV:LCV).LE.'z'))
     &          STRING(LCV:LCV) = CHAR(ICHAR(STRING(LCV:LCV))-32)
            IF (STRING(LCV:LCV).LT.' ') STRING(LCV:LCV) = ' '
10      CONTINUE
        RETURN
        END
 
        INTEGER FUNCTION IN_LIST( NUM, DESC, CODE, TOTAL, CURR )
******* This function returns an integer value in one of three states.
*******     -99 = The proj/desc passed in is blank.
*******      -1 = It's not in the list
*******    +xxx = This proj/desc if found at index xxx in the arrays.
******* CURR is passed in as the index of the proj/desc being checked
******* by this call.
        INTEGER TOTAL, P_CODE(0:200), I, NCNT, CURR, FIRST
        CHARACTER P_NUM(0:200)*7, P_DESC(0:200)*20
        CHARACTER NAMES(0:10)*6, P_HRS(0:6,0:200)*5, CTRL(0:200)
        CHARACTER NUM*7, DESC*20, CODE*5
        COMMON/DATA/P_NUM,P_DESC,P_CODE,P_HRS,CTRL,FIRST
        COMMON/NAMES/NAMES,NCNT
        IN_LIST = -1
        IF ((NUM(1:6).EQ.'      ') .AND.
     &      (DESC.EQ.'                    ')) THEN
            IN_LIST = -99
            RETURN
        ENDIF
        DO 10, I=0,TOTAL-1
            IF (I.NE.CURR) THEN
                IF ((((P_NUM(I)(:6).EQ.NUM(:6)) .OR.
     &                (NUM(:6).EQ.'      ')) .AND.
     &              ((P_DESC(I).EQ.DESC).OR.(DESC(1:3).EQ.'   '))).AND.
     &              ((NAMES(P_CODE(I)).EQ.CODE) .OR.
     &              (P_CODE(I).EQ.0))) THEN
                    IN_LIST = I
                    RETURN
                ENDIF
            ENDIF
10      CONTINUE
        RETURN
        END
 
        SUBROUTINE VALIDATE( NUM, DESC )
******* This routine pulls out valid matches to the proj/desc passed
******* in from the database.
        CHARACTER NUM*7, DESC*20, LIST(0:300)*37, VAL_PROJ*150
        CHARACTER BGN_LIST_4*35, MODE_IN*2, NAMES(0:10)*6
        INTEGER LCNT, NCNT, I, Q
        INTEGER ECOD,CP6E,OPER,OVRW,EXST,SKIP,ARGN
        COMMON/ERRS/ECOD,CP6E,OPER,OVRW,EXST,SKIP,ARGN
        COMMON/NAMES/NAMES,NCNT
        COMMON/DATA2/LIST,LCNT
        MODE_IN         = 'IN'
        BGN_LIST_4      = 'MODE=IN,SHARE=IN,PSA(SHARE=ALL)'
        VAL_PROJ        = 'SEL PROJECT,DESCRIPTION,METRICS FROM '//
     &      'PSA WHERE PROJECT IS LIKE #1 OR DESCRIPTION IS LIKE #2'//
     &      ' AND STATUS IS IN(''A'',''I'')'
        DO 10, I=0,300
            LIST(I) = ' '
10      CONTINUE
        LCNT = -1
        CALL ARES_BEGIN( BGN_LIST_4 )
        IF (ECOD.NE.0) RETURN
        CALL ARES_STARTEXT( VAL_PROJ, Q, MODE_IN, NUM(1:6), DESC )
        IF (ECOD.NE.0) RETURN
        LCNT = 0
        CALL ARES_FETCH(Q,LIST(0)(1:7),LIST(0)(10:29),LIST(0)(32:32))
            LCNT = LCNT + 1
        REPEAT 30, WHILE (ECOD.EQ.0)
            CALL ARES_FETCH( Q, LIST(LCNT)(1:7), LIST(LCNT)(10:29),
     &          LIST(LCNT)(32:32) )
                LCNT = LCNT + 1
30      CONTINUE
        LCNT = LCNT - 1
        CALL ARES_RELEASE( Q )
        CALL ARES_COMMIT
        RETURN
        END
 
        SUBROUTINE LOAD_LIST( FIRST )
******* This routine loads the Info form with the LIST array elements
******* starting with FIRST.  It will first shut off the rev-vid in
******* the Cmd field, then load the data into the Info fields.  If
******* a -1 is passed in, the Cmd and Info fields are removed from
******* the form.
        INTEGER FIRST, LCNT, I
        CHARACTER LIST(0:300)*37
        COMMON/DATA2/LIST,LCNT
        INCLUDE XH_FORGE_FORTRAN
        INCLUDE *VOUCHER_CF2
        CALL FORMZAPS( FORM2 )
        DO 20, I=0,16
            CMD2_S(I) = 1
            CMD2(I) = ' '
20      CONTINUE
        CALL FORMMDFY(FORM2,FORGE_ABSNONE,FORM2_S)
        CALL FORMWRITE(FORM2,FORM2_S)
        CALL FORMCURRENT(FORM2)
        CALL FORMZAPS( FORM2 )
        IF (FIRST.EQ.-1) THEN
            DO 10, I=0,16
                INFO2(I) = ' '
                INFO2_S(I) = 1
10          CONTINUE
        ELSE
            DO 30, I=FIRST,MIN(FIRST+16,LCNT-1)
                INFO2(I-FIRST) = LIST(I)
                INFO2_S(I-FIRST) = 1
                IF (INFO2(I-FIRST)(32:32).EQ.'Y')
     &              INFO2(I-FIRST)(32:37)='Metric'
30          CONTINUE
        ENDIF
        CALL FORMWRITE(FORM2,FORM2_S)
        CALL FORMCURRENT(FORM2)
        CALL FORMZAPS( FORM2 )
        RETURN
        END

        SUBROUTINE VERIFY_WEEK_HOURS( HOURS, HNUM, ERR_TEXT )
        INCLUDE *VOUCHER_CF1
        CHARACTER HOURS*5, ERR_TEXT*79
        INTEGER HNUM, ARYINDX, IDAY, CMAX, DEC
        ERR_TEXT = ' '
        CALL UPCASE(HOURS,5)
        ARYINDX = (HNUM-HOURS1_N) / 7
        IDAY = HNUM - (ARYINDX*7) - HOURS1_N + 1
        IF (HOURS(1:1).EQ.'?') THEN
            HOURS(1:1) = ' '
            HOURS1(IDAY-1,ARYINDX)(1:1) = ' '
            CALL FORMWRITE(FORM1,HOURS1(IDAY-1,ARYINDX))
            IF (ACT1(ARYINDX)(1:1).EQ.' ') THEN
                ERR_TEXT = 'Enter hours worked (in quarter hour '//
     &              'increments) for '//
     &              DESC1(ARYINDX)
            ELSE
                ERR_TEXT = 'Enter hours worked (in quarter hour '//
     &              'increments) for '//DESC1(ARYINDX)//' '//
     &              ACT1(ARYINDX)
            ENDIF
            RETURN
        ENDIF
        DO 10, I=1,5
            IF (INDEX(' .0123456789',HOURS(I:I)).EQ.0) THEN
                ERR_TEXT = 'Invalid characters in hours field.  No '//
     &              'Control Characters allowed in Week mode.'
                RETURN
            ENDIF
10      CONTINUE
        DEC = INDEX(HOURS,'.')
        IF ((DEC.GT.0).AND.(INDEX(HOURS(DEC+1:),'.').NE.0)) THEN
            ERR_TEXT = 'Too many decimal points.  Please re-enter.'
            RETURN
        ENDIF
        READ(HOURS,'(BN,F5.2)')DHOURS
        CHANGED = .TRUE.
        IF (DHOURS.EQ.0.0) THEN
            HOURS = '  .  '
            RETURN
        ENDIF
        IF (INDEX(HOURS,'.').EQ.0) DHOURS=DHOURS*100
        IF (DHOURS.GT.99) THEN
!           ERR_TEXT = 'Gimme a break!  There are only 24 hours in '//
!    &          'a day!'
            ERR_TEXT = 'This number is too big to be displayed.  '//
     &          'Please enter a number less than 99.'
            RETURN
        ENDIF
         WRITE(HOURS,'(F5.2)')DHOURS
        IF ((HOURS(4:5).NE.'00').AND.(HOURS(4:5).NE.'25').AND.
     &      (HOURS(4:5).NE.'50').AND.(HOURS(4:5).NE.'75')) THEN
            ERR_TEXT = 'Only quarter hour increments are allowed. '//
     &          ' ( .00, .25, .50, .75 )'
            RETURN
        ENDIF
        RETURN
        END
